successful_murder_effect = { #character scope
	change_variable = { which = murder_count value = 1 }
}

check_notoriety_level_effect = {

#province scope
#Checks notoriety_value and sets the notorious_* level appropriately
#Level thresholds are and should be only declared in this effect!

	if = {
		limit = { #Provinces with low notoriety_value and no notorious_* level (supposedly the majority) will not be checked further
			OR = {
				check_variable = { which = notoriety_value value = 100 } #NotorietyLvlDef1
				has_province_modifier = notorious_1
				has_province_modifier = notorious_2
				has_province_modifier = notorious_3
			}
		}
		
		if = { #If notorious_* level is lower than expected
			limit = {
				OR = {
					AND = {
						check_variable = { which = notoriety_value value = 100 } #NotorietyLvlDef1
						NOR = {
							has_province_modifier = notorious_1
							has_province_modifier = notorious_2
							has_province_modifier = notorious_3
						}
					}
					AND = {
						check_variable = { which = notoriety_value value = 250 } #NotorietyLvlDef2
						NOR = {
							has_province_modifier = notorious_2
							has_province_modifier = notorious_3
						}
					}
					AND = {
						check_variable = { which = notoriety_value value = 500 } #NotorietyLvlDef3
						NOT = {
							has_province_modifier = notorious_3
						}
					}
				}
			}
			if = {
				limit = {
					check_variable = { which = notoriety_value value = 100 } #NotorietyLvlDef1
					NOR = {
						has_province_modifier = notorious_1
						has_province_modifier = notorious_2
						has_province_modifier = notorious_3
					}
				}
				add_province_modifier = { name = notorious_1 duration = -1 }
			}
			if = {
				limit = {
					check_variable = { which = notoriety_value value = 250 } #NotorietyLvlDef2
					has_province_modifier = notorious_1
				}
				add_province_modifier = { name = notorious_2 duration = -1 }
				remove_province_modifier = notorious_1
			}
			if = {
				limit = {
					check_variable = { which = notoriety_value value = 500 } #NotorietyLvlDef3
					has_province_modifier = notorious_2
				}
				add_province_modifier = { name = notorious_3 duration = -1 }
				remove_province_modifier = notorious_2
			}
			province_event = { id = MMI.2 }
		}
		
		else_if = { #If notorious_* level is higher than expected
			limit = {
				OR = {
					AND = {
						has_province_modifier = notorious_3
						NOT = {
							check_variable = { which = notoriety_value value = 500 } #NotorietyLvlDef3
						}
					}
					AND = {
						OR = {
							has_province_modifier = notorious_2
							has_province_modifier = notorious_3
						}
						NOT = {
							check_variable = { which = notoriety_value value = 250 } #NotorietyLvlDef2
						}
					}
					AND = {
						OR = {
							has_province_modifier = notorious_1
							has_province_modifier = notorious_2
							has_province_modifier = notorious_3
						}
						NOT = {
							check_variable = { which = notoriety_value value = 100 } #NotorietyLvlDef1
						}
					}
				}
			}
			if = {
				limit = {
					NOT = {
						check_variable = { which = notoriety_value value = 500 } #NotorietyLvlDef3
					}
					has_province_modifier = notorious_3
				}
				add_province_modifier = { name = notorious_2 duration = -1 }
				remove_province_modifier = notorious_3
			}
			if = {
				limit = {
					NOT = {
						check_variable = { which = notoriety_value value = 250 } #NotorietyLvlDef2
					}
					has_province_modifier = notorious_2
				}
				add_province_modifier = { name = notorious_1 duration = -1 }
				remove_province_modifier = notorious_2
			}
			if = {
				limit = {
					NOT = {
						check_variable = { which = notoriety_value value = 100 } #NotorietyLvlDef1
					}
					has_province_modifier = notorious_1
				}
				remove_province_modifier = notorious_1
			}
			province_event = { id = MMI.3 }
		}
	}
}
