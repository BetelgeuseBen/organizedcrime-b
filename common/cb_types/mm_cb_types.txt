			limit = {

				}

						who = PREV
						modifier = won_tributary_war

					}
				}
			}

	}

		ROOT = {
			show_scope_change = no
			prestige = -100
		}
		FROM = {
			show_scope_change = no
			prestige = 100
		}
	}

		ROOT = {
			show_scope_change = no

				to = FROM
				value = 4.0
			}
		}

		}
	}

	on_fail_title = {

		}
