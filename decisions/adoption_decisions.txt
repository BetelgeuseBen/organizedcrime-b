targetted_decisions = {

		is_high_prio = yes
		ai_check_interval = 12
		only_playable = yes
		filter = home_court
		
		from_potential = {
			has_character_flag = character_adopting
		}
		potential = {
			is_ruler = no
			is_heir = no
			prisoner = no
			is_incapable = no
			NOT = { dynasty = FROM }
			NOT = { trait = adopted }
			NOT = { is_close_relative = FROM }
		}
		allow = {
			FROM = { prestige = 100 }
			opinion = { who = FROM value = 80 }
			#reverse_opinion = { who = FROM value = 80 }
		}
		effect = {
			FROM = {
				prestige = -100

			}
			dynasty = FROM
			add_trait = adopted
			recalc_succession = yes
			opinion = { modifier = opinion_adoption who = FROM }
			reverse_opinion = { modifier = opinion_adoption who = FROM }
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 1
		}
	}
