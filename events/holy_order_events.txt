# NYPD succession - duchy holder is always Deputy Chief, fired from on_new_holder, on_new_holder_inheritance, on_new_holder_usurpation
character_event = {
	id = MMHO.5
	hide_window = yes

	is_triggered_only = yes
	only_playable = yes
	
	trigger = {
		is_government_potential = police_government
		primary_title = { holy_order = yes }
	}
	immediate = {
		if = {
			limit = { NOT = { society_member_of = new_york_police_department } }
			leave_society = yes
			join_society = new_york_police_department
		}
		if = {
			limit = {
				society_rank = {
					rank == 1
					society = new_york_police_department
				}
			}
			society_rank_up = {
				rank = 3
				society = new_york_police_department
			}
		}
		if = {
			limit = {
				society_rank = {
					rank == 2
					society = new_york_police_department
				}
			}
			society_rank_up = {
				rank = 2
				society = new_york_police_department
			}
		}
		if = {
			limit = {
				society_rank = {
					rank == 3
					society = new_york_police_department
				}
			}
			society_rank_up = {
				rank = 1
				society = new_york_police_department
			}
		}
		any_courtier = {
			if = {
				limit = {
					is_in_society = no
					is_adult = yes
					prisoner = no
					NOT = { trait = incapable }
				}
			join_society = new_york_police_department
			}
		}
		any_vassal = {
			if = {
				limit = {
					is_in_society = no
					is_adult = yes
					prisoner = no
					NOT = { trait = incapable }
				}
			join_society = new_york_police_department
			}
		}
		FROMFROM = {
			society_rank_down = 1
			any_courtier = {
				if = {
					limit = {
						is_in_society = no
						is_adult = yes
						prisoner = no
						NOT = { trait = incapable }
					}
				join_society = new_york_police_department
				}
			}
			any_vassal = {
				if = {
					limit = {
						is_in_society = no
						is_adult = yes
						prisoner = no
						NOT = { trait = incapable }
					}
				join_society = new_york_police_department
				}
			}
		}
	}
}

# Character joining NYPD effects
character_event = {
	id = MMHO.6
	hide_window = yes

	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				has_minor_title = title_commander
			}
			remove_title = title_commander
		}
		if = {
			limit = {
				has_minor_title = title_appointed_regent
			}
			remove_title = title_appointed_regent
		}
		if = {
			limit = {
				has_minor_title = title_patrician_heir
			}
			remove_title = title_patrician_heir
		}
		clear_wealth = event_target:order_target
		move_character = event_target:order_target
		end_inaccessibility_effect = yes #removes in_hiding or in_seclusion
		leave_society = yes
		join_society = new_york_police_department
	}
}
